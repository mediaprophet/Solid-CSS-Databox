    Interface PodStore

    Can be used to create new pods and find relevant information. Also keeps track of the owners of a pod. The visible parameter indicates if an owner should be publicly exposed or not.

    interface PodStore {
        create: (
            accountId: string,
            settings: PodSettings,
            overwrite: boolean,
        ) => Promise<string>;
        findByBaseUrl: (
            baseUrl: string,
        ) => Promise<{ accountId: string; id: string } | undefined>;
        findPods: (accountId: string) => Promise<{ baseUrl: string; id: string }[]>;
        get: (
            id: string,
        ) => Promise<{ accountId: string; baseUrl: string } | undefined>;
        getOwners: (
            id: string,
        ) => Promise<{ visible: boolean; webId: string }[] | undefined>;
        removeOwner: (id: string, webId: string) => Promise<void>;
        updateOwner: (id: string, webId: string, visible: boolean) => Promise<void>;
    }
